# CODING_GUIDELINES
- **프로젝트명**: 오늘득템 (중고거래 플랫폼)
- **기술 스택**: Spring Boot, JPA, Thymeleaf, Oracle DB
- **아키텍처**: MVC (Model2) 패턴

## 🏗️ 프로젝트 구조

```
src/main/java/dtem/
├── config/          # 설정 클래스 (WebSocket, Security 등)
├── controller/      # 컨트롤러 (웹 요청 처리)
├── dao/            # 데이터 접근 객체 (Repository 래퍼)
├── dto/            # 데이터 전송 객체
├── entity/         # JPA 엔티티 (데이터베이스 테이블 매핑)
├── repository/     # JPA Repository 인터페이스
└── service/        # 비즈니스 로직 서비스
```

2. 테이블, 시퀀스 추가

```sql
------------------------------------------------------------
-- USERS 테이블 (회원 정보)
------------------------------------------------------------
CREATE TABLE users (
    userid NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  -- 자동 증가
    email VARCHAR2(100) UNIQUE NOT NULL,        -- 이메일 (로그인 ID, 중복 불가)
    password VARCHAR2(100) NOT NULL,            -- 비밀번호
    username VARCHAR2(50) NOT NULL,             -- 이름
    phone VARCHAR2(20),                          -- 전화번호 (중복 가능)
    profileimage VARCHAR2(500),                 -- 프로필 이미지 (/uploads/profiles/uuid_image.jpg)
    addr1 VARCHAR2(100),                        -- 우편번호
    addr2 VARCHAR2(200),                        -- 도로명 주소
    addr3 VARCHAR2(200),                        -- 지번 주소
    addr4 VARCHAR2(200),                        -- 상세 주소
    introduce VARCHAR2(500),                    -- 자기소개
    rating NUMBER(3,2) DEFAULT 0.00,            -- 거래평점
    reviewcount NUMBER DEFAULT 0,               -- 받은 리뷰 개수
    sellcount NUMBER DEFAULT 0,                 -- 판매 완료 횟수
    buycount NUMBER DEFAULT 0,                  -- 구매 완료 횟수
    createdat TIMESTAMP DEFAULT SYSTIMESTAMP    -- 회원가입 일시
);

COMMENT ON TABLE users IS '회원 정보';
COMMENT ON COLUMN users.email IS '이메일 (로그인 ID, 중복 불가)';
COMMENT ON COLUMN users.phone IS '전화번호 (중복 가능)';
COMMENT ON COLUMN users.profileimage IS '프로필 이미지 경로';

------------------------------------------------------------
-- POSTS 테이블 (게시물)
------------------------------------------------------------
CREATE TABLE posts (
    postid NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    userid NUMBER NOT NULL,
    title VARCHAR2(200) NOT NULL,
    content VARCHAR2(4000) NOT NULL,
    postimage VARCHAR2(500), -- 여러 이미지 경로를 쉼표로 구분하여 저장 (최대 10개)
    price NUMBER DEFAULT 0 NOT NULL,
    category VARCHAR2(50) NOT NULL,
    location VARCHAR2(100) NOT NULL,
    conditionstatus VARCHAR2(20),
    trademethod VARCHAR2(50),
    negotiable CHAR(1) DEFAULT 'N',
    status VARCHAR2(20) DEFAULT 'SELLING',
    hit NUMBER DEFAULT 0,
    wishlistcount NUMBER DEFAULT 0,
    createdat TIMESTAMP DEFAULT SYSTIMESTAMP,
    updatedat TIMESTAMP,
    CONSTRAINT fk_post_user FOREIGN KEY (userid) 
        REFERENCES users(userid) ON DELETE CASCADE
);

CREATE INDEX idx_post_createdat ON posts(createdat DESC);
CREATE INDEX idx_post_wishlistcount ON posts(wishlistcount DESC);
CREATE INDEX idx_post_category ON posts(category);
CREATE INDEX idx_post_location ON posts(location);
CREATE INDEX idx_post_status ON posts(status);

COMMENT ON TABLE posts IS '중고거래 게시물';

------------------------------------------------------------
-- POST_LIKE 테이블 (찜/좋아요)
------------------------------------------------------------
CREATE TABLE post_like (
    likeseq NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    postid NUMBER NOT NULL,
    userid NUMBER NOT NULL,
    createdat TIMESTAMP DEFAULT SYSTIMESTAMP,
    CONSTRAINT fk_like_post FOREIGN KEY (postid) 
        REFERENCES posts(postid) ON DELETE CASCADE,
    CONSTRAINT fk_like_user FOREIGN KEY (userid) 
        REFERENCES users(userid) ON DELETE CASCADE,
    CONSTRAINT uq_post_like UNIQUE (postid, userid)
);

COMMENT ON TABLE post_like IS '게시물 찜(좋아요)';

------------------------------------------------------------
-- 채팅 관련 테이블 (WebSocket 실시간 채팅)
------------------------------------------------------------
-- 시퀀스
CREATE SEQUENCE CHAT_MESSAGE_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;
CREATE SEQUENCE CHAT_ROOM_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;
CREATE SEQUENCE TRADE_SEQ START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

-- 채팅 메시지 (소문자 + 큰따옴표 주의!)
CREATE TABLE "chat_message" (
    "id" NUMBER(19) PRIMARY KEY,
    "sender_id" VARCHAR2(255) NOT NULL,
    "receiver_id" VARCHAR2(255) NOT NULL,
    "room_id" VARCHAR2(255) NOT NULL,        -- 채팅방 ID (상품별 구분)
    "content" VARCHAR2(2000) NOT NULL,
    "type" VARCHAR2(50),
    "timestamp" TIMESTAMP NOT NULL,
    "is_read" NUMBER(1) DEFAULT 0 NOT NULL
);

-- 채팅방
CREATE TABLE "chat_room" (
    "id" NUMBER(19) PRIMARY KEY,
    "room_id" VARCHAR2(255) UNIQUE NOT NULL,
    "user1_id" VARCHAR2(255) NOT NULL,
    "user2_id" VARCHAR2(255) NOT NULL,
    "post_id" NUMBER(19),
    "created_at" TIMESTAMP NOT NULL,
    "last_message_at" TIMESTAMP,
    "last_message" VARCHAR2(2000),
    "user1_left" NUMBER(1) DEFAULT 0,      -- user1 나가기 여부 (0: 미나감, 1: 나감)
    "user2_left" NUMBER(1) DEFAULT 0       -- user2 나가기 여부 (0: 미나감, 1: 나감)
);

-- 거래 (채팅방과 연동)
CREATE TABLE "trade" (
    "trade_id" NUMBER(19) PRIMARY KEY,
    "post_id" NUMBER(19) NOT NULL,
    "seller_id" VARCHAR2(255) NOT NULL,
    "buyer_id" VARCHAR2(255) NOT NULL,
    "room_id" VARCHAR2(255) NOT NULL,
    "status" VARCHAR2(50) NOT NULL,
    "created_at" TIMESTAMP NOT NULL,
    "completed_at" TIMESTAMP
);

-- 리뷰 (거래 완료 후 작성)
CREATE TABLE "review" (
    "review_id" NUMBER(19) PRIMARY KEY,
    "trade_id" NUMBER(19) NOT NULL,
    "reviewer_id" VARCHAR2(255) NOT NULL,        -- 리뷰 작성자
    "reviewee_id" VARCHAR2(255) NOT NULL,        -- 리뷰 받는 사람
    "response_speed" NUMBER(1),                  -- 응답속도 (1-5)
    "rating" NUMBER(1) NOT NULL,                 -- 별점 (1-5)
    "manner_score" NUMBER(1),                    -- 매너지수 (1-5)
    "will_trade_again" CHAR(1),                  -- 재거래희망 (Y/N)
    "review_content" VARCHAR2(2000),             -- 리뷰 내용
    "created_at" TIMESTAMP NOT NULL
);

-- 리뷰 시퀀스
CREATE SEQUENCE review_seq START WITH 1 INCREMENT BY 1;

-- 채팅 및 리뷰 인덱스
CREATE INDEX idx_chat_message_sender ON "chat_message"("sender_id");
CREATE INDEX idx_chat_message_receiver ON "chat_message"("receiver_id");
CREATE INDEX idx_chat_message_room_id ON "chat_message"("room_id");
CREATE INDEX idx_chat_message_timestamp ON "chat_message"("timestamp" DESC);
CREATE INDEX idx_chat_room_user1 ON "chat_room"("user1_id");
CREATE INDEX idx_chat_room_user2 ON "chat_room"("user2_id");
CREATE INDEX idx_chat_room_post ON "chat_room"("post_id");
CREATE INDEX idx_chat_room_roomid ON "chat_room"("room_id");
CREATE INDEX idx_trade_room ON "trade"("room_id");
CREATE INDEX idx_trade_post ON "trade"("post_id");
CREATE INDEX idx_review_trade ON "review"("trade_id");
CREATE INDEX idx_review_reviewer ON "review"("reviewer_id");
CREATE INDEX idx_review_reviewee ON "review"("reviewee_id");

-- 제약조건
ALTER TABLE "trade" ADD CONSTRAINT chk_trade_status 
    CHECK ("status" IN ('IN_PROGRESS', 'COMPLETED', 'CANCELLED'));

ALTER TABLE "review" ADD CONSTRAINT chk_review_response_speed
    CHECK ("response_speed" BETWEEN 1 AND 5 OR "response_speed" IS NULL);

ALTER TABLE "review" ADD CONSTRAINT chk_review_rating
    CHECK ("rating" BETWEEN 1 AND 5);

ALTER TABLE "review" ADD CONSTRAINT chk_review_manner_score
    CHECK ("manner_score" BETWEEN 1 AND 5 OR "manner_score" IS NULL);

ALTER TABLE "review" ADD CONSTRAINT chk_review_will_trade_again
    CHECK ("will_trade_again" IN ('Y', 'N') OR "will_trade_again" IS NULL);

-- 코멘트
COMMENT ON TABLE "chat_message" IS '채팅 메시지 (상품별 채팅방 구분)';
COMMENT ON COLUMN "chat_message"."room_id" IS '채팅방 ID (상품별로 구분된 채팅방)';
COMMENT ON TABLE "chat_room" IS '채팅방';
COMMENT ON TABLE "trade" IS '거래 정보';
COMMENT ON TABLE "review" IS '거래 후 작성하는 리뷰';
COMMENT ON COLUMN "review"."reviewer_id" IS '리뷰 작성자 ID';
COMMENT ON COLUMN "review"."reviewee_id" IS '리뷰 받는 사람 ID';
COMMENT ON COLUMN "review"."response_speed" IS '응답속도 (1-5)';
COMMENT ON COLUMN "review"."rating" IS '별점 (1-5)';
COMMENT ON COLUMN "review"."manner_score" IS '매너지수 (1-5)';
COMMENT ON COLUMN "review"."will_trade_again" IS '재거래희망 (Y/N)';
```
